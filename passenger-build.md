
## Building the Nginx module
apt-get build-dep nginx
apt-get source nginx
apt-get install ruby rake libcurl4-openssl-dev ruby-dev


wget https://www.phusionpassenger.com/latest_stable_tarball
tar -xvf latest_stable_tarball


add to debian/rules at the end of common_configure_flags (make sure to add \ above)
	--add-dynamic-module=/tmp/build-nginx/passenger-6.0.4/src/nginx_module


dpkg-buildpackage -us -uc


## But it is incomplete, we need passenger server parts

env NOEXEC_DISABLE=1 #{PlatformInfo.rake_command} nginx:clean nginx RELEASE=yes
